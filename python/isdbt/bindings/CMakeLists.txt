########################################################################
# Check if there is C++ code at all
########################################################################
if(NOT isdbt_sources)
    MESSAGE(STATUS "No C++ sources... skipping python bindings")
    return()
endif(NOT isdbt_sources)

########################################################################
# Check for pygccxml
########################################################################
GR_PYTHON_CHECK_MODULE_RAW(
    "pygccxml"
    "import pygccxml"
    PYGCCXML_FOUND
    )

include(GrPybind)

########################################################################
# Python Bindings
########################################################################

list(APPEND isdbt_python_files
    python_bindings.cc)

GR_PYBIND_MAKE_OOT(isdbt
   ../../..
   gr::isdbt
   "${isdbt_python_files}")

# copy bindings extension for use in QA test module
add_custom_command(TARGET isdbt_python POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy $<TARGET_FILE:isdbt_python>
    ${PROJECT_BINARY_DIR}/test_modules/gnuradio/isdbt/
)

install(TARGETS isdbt_python DESTINATION ${GR_PYTHON_DIR}/gnuradio/isdbt COMPONENT pythonapi)
